// Handle sorting by timeline or hashtags
const sort = document.getElementById('sort');
const posts = Array.from(document.querySelectorAll('.post'));

sort.addEventListener('change', function() {
const selectedValue = this.value;

if (selectedValue === 'timeline') {
posts.sort((a, b) => a.dataset.timeline - b.dataset.timeline);
} else if (selectedValue === 'hashtags') {
posts.sort((a, b) => a.dataset.hashtag.localeCompare(b.dataset.hashtag));
}

// Clear and append sorted posts to blog section
const blogSection = document.querySelector('.blog-section');
posts.forEach(post => blogSection.appendChild(post));
});

// Simulate the gritty to polished design transition
window.addEventListener('scroll', function() {
if (window.scrollY > 300) {
document.body.classList.add('transformed');
} else {
document.body.classList.remove('transformed');
}
});
</script>
